CREATE TABLE `BibliotecaMateriais` (
  `codMaterial` bigint(20) unsigned NOT NULL auto_increment,
  `codTurma` bigint(20) unsigned NOT NULL default '0',
  `titulo` varchar(80) NOT NULL default '',
  `autor` varchar(80) NOT NULL default '',
  `palavras` text,
  `codUsuario` bigint(20) unsigned NOT NULL default '0',
  `tipoMaterial` char(1) NOT NULL default '',
  `material` text,
  `tamanhoMaterial` bigint(20) unsigned default NULL,
  `data` date NOT NULL default '0000-00-00',
  `hora` time NOT NULL default '00:00:00',
  `numeroComentarios` int(11) NOT NULL default '0',
  `intergrupo` char(1) NOT NULL default '',
  PRIMARY KEY  (`codMaterial`)
) ENGINE=MyISAM DEFAULT CHARSET=latin1;

CREATE TABLE `BibliotecaComentariosVistos` (
  `codMaterial` bigint(20) NOT NULL default '0',
  `codComentario` bigint(20) NOT NULL default '0',
  `codUsuario` bigint(20) NOT NULL default '0',
  PRIMARY KEY  (`codMaterial`,`codComentario`,`codUsuario`)
) ENGINE=MyISAM DEFAULT CHARSET=latin1;

CREATE TABLE `BibliotecaComentarios` (
  `codComentario` bigint(20) unsigned NOT NULL auto_increment,
  `codMaterial` bigint(20) unsigned NOT NULL default '0',
  `codUsuario` bigint(20) unsigned NOT NULL default '0',
  `comentario` text NOT NULL,
  `data` date NOT NULL default '0000-00-00',
  `hora` date NOT NULL default '0000-00-00',
  PRIMARY KEY  (`codComentario`)
) ENGINE=MyISAM DEFAULT CHARSET=latin1;
